จงหาผลลัพธ์ทั้งหมดที่เกิดจากคำสั่ง console.log

```js
console.log('' + 1 + 0);
console.log('' - 1 + 0));
console.log(true + false);
console.log(6 / '3');
console.log('2' \* '3');
console.log(' -9 ' - 5);
console.log(' -9 ' + 5);
console.log('4px' - 2);
console.log(2 + 4 + 'px');
console.log(null + 1);
console.log(undefined + 1);
```
